+def timed(fn: Callable) -> Callable:
+    """Accumulate wall-clock time of ``fn`` on ``fn.total_time`` / ``fn.n_calls``.
+
+    Kept side-effect free for the caller (returns the original value) so it can wrap
+    hot functions during profiling without changing behaviour. ``functools.wraps``
+    preserves the wrapped function's name / docstring / signature metadata.
+    """
+    @functools.wraps(fn)
+    def wrapper(*args, **kwargs):
+        t0 = time.perf_counter()
+        try:
+            return fn(*args, **kwargs)
+        finally:
+            wrapper.total_time += time.perf_counter() - t0
+            wrapper.n_calls += 1
+    wrapper.total_time = 0.0
+    wrapper.n_calls = 0
+    return wrapper
+

+def retry(times: int = 3, exceptions: tuple = (Exception,), base_delay: float = 0.5,
+          backoff: float = 2.0):
+    """Retry ``fn`` up to ``times`` attempts with exponential backoff.
+
+    Parameterized decorator for unstable I/O (the OpenAI API call below). The final
+    failure is re-raised unchanged so callers still see the real exception; the
+    wrapped function's signature/metadata is preserved with ``functools.wraps``.
+    """
+    def deco(fn: Callable) -> Callable:
+        @functools.wraps(fn)
+        def wrapper(*args, **kwargs):
+            delay = base_delay
+            last = None
+            for attempt in range(1, times + 1):
+                try:
+                    return fn(*args, **kwargs)
+                except exceptions as e:  # noqa: PERF203
+                    last = e
+                    if attempt == times:
+                        raise
+                    time.sleep(delay)
+                    delay *= backoff
+            raise last  # pragma: no cover
+        return wrapper
+    return deco
